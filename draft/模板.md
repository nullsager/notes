## 字典树
```cpp
vector<array<int, 26>> tree(m * 20);  // 20是字符串最大长度，m是操作次数
vector<int> ed(m * 20);
vector<int> pass(m * 20);
int cnt = 1;
auto myinsert = [&](string_view word) {
  int cur = 1;
  pass[cur]++;
  for (char c : word) {
    int x = c - 'a';
    if (tree[cur][x] == 0) {
      tree[cur][x] = ++cnt;
    }
    cur = tree[cur][x];
    pass[cur]++;
  }
  ed[cur]++;
};
auto mysearch = [&](string_view word) {
  int cur = 1;
  for (char c : word) {
    int x = c - 'a';
    if (tree[cur][x] == 0) {
      return 0;
    }
    cur = tree[cur][x];
  }
  return ed[cur];
};
auto prefixNumber = [&](string_view word) {
  int cur = 1;
  for (char c : word) {
    int x = c - 'a';
    if (tree[cur][x] == 0) {
      return 0;
    }
    cur = tree[cur][x];
  }
  return pass[cur];
};
auto mydelete = [&](string_view word) {
  if (mysearch(word) > 0) {
    int cur = 1;
    pass[cur]--;
    for (char c : word) {
      int x = c - 'a';
      if (--pass[tree[cur][x]] == 0) {
        tree[cur][x] = 0;
        return;
      }
      cur = tree[cur][x];
    }
    ed[cur]--;
  }
};
```

## 等差数列差分

```cpp
void set(int l, int r, int s, int e, int d) {
  arr[l] += s;
  arr[l + 1] += d - s;
  arr[r + 1] -= d + e;
  arr[r + 2] += e;
}
void build() {
  for (int i = 1; i <= n; i++) {
    arr[i] += arr[i - 1];
  }
  for (int i = 1; i <= n; i++) {
    arr[i] += arr[i - 1];
  }
}
```

## 二维差分

```cpp
void add(int a, int b, int c, int d, int k) {
  diff[a][b] += k;
  diff[c + 1][b] -= k;
  diff[a][d + 1] -= k;
  diff[c + 1][d + 1] += k;
}

void build() {
  for (int i = 1; i <= n; i++) {
    for (int j = 1; j <= n; j++) {
      diff[i][j] += diff[i - 1][j] + diff[i][j - 1] - diff[i - 1][j - 1];
    }
  }
}
```

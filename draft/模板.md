## 字典树
```cpp
vector<array<int, 26>> tree(m * 20);  // 20是字符串最大长度，m是操作次数
vector<int> ed(m * 20);
vector<int> pass(m * 20);
int cnt = 1;
auto insert = [&](string_view word) {
  int cur = 1;
  pass[cur]++;
  for (char c : word) {
    int x = c - 'a';
    if (tree[cur][x] == 0) {
      tree[cur][x] = ++cnt;
    }
    cur = tree[cur][x];
    pass[cur]++;
  }
  ed[cur]++;
};
auto mysearch = [&](string_view word) {
  int cur = 1;
  for (char c : word) {
    int x = c - 'a';
    if (tree[cur][x] == 0) {
      return 0;
    }
    cur = tree[cur][x];
  }
  return ed[cur];
};
auto prefixNumber = [&](string_view word) {
  int cur = 1;
  for (char c : word) {
    int x = c - 'a';
    if (tree[cur][x] == 0) {
      return 0;
    }
    cur = tree[cur][x];
  }
  return pass[cur];
};
auto mydelete = [&](string_view word) {
  if (mysearch(word) > 0) {
    int cur = 1;
    pass[cur]--;
    for (char c : word) {
      int x = c - 'a';
      if (--pass[tree[cur][x]] == 0) {
        tree[cur][x] = 0;
        return;
      }
      cur = tree[cur][x];
    }
    ed[cur]--;
  }
};
```

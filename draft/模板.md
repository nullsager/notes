```cpp
auto myinsert = [&](string_view word) {

  pass[cur]++;
  for (char c : word) {
    int x = c - 'a';
    if (tree[cur][x] == 0) tree[cur][x] = ++cnt;
    cur = tree[cur][x];
    pass[cur]++;
  }

};

  int cur = 1;
  for (char c : word) {

    if (tree[cur][x] == 0) return 0;
    cur = tree[cur][x];

  int cur = 1;
```

## 滑动窗口

给定一个含有 n 个正整数的数组和一个正整数 target 。找出该数组中满足其总和大于等于 target 的长度最小的子数组。

```cpp
int minSubArrayLen(int target, vector<int>& nums) {
  int ans = INT_MAX;
  for (int l = 0, r = 0, sum = 0; r < nums.size(); r++) {
    sum += nums[r];
    while (sum - nums[l] >= target) sum -= nums[l++];
    if (sum >= target) ans = min(ans, r - l + 1);
  }
  return ans == INT_MAX ? 0 : ans;
}
```

```cpp
int lengthOfLongestSubstring(string s) {
  int ans = 0;
  vector<int> cnt(128, 0);
  int l = 0;
  for (int r = 0; r < s.size(); r++) {
    cnt[s[r]]++;  // 将当前字符加入窗口

  }
  return ans;
}
```

给定两个字符串 s 和 t，长度分别是 m 和 n，返回 s 中的最短窗口子串，使得该子串包含 t 中的每一个字符（包括重复字符）。如果没有这样的子串，返回空字符串 ""

```cpp
string minWindow(string s, string t) {
  unordered_map<char, int> mp;
  for (char c : t) mp[c]++;
  int k = 0;  // 窗口中已经包含的t的字符数量
  unordered_map<char, int> cnt;
  int res = INT_MAX;
  int ll = 0, rr = INT_MAX;
  for (int l = 0, r = 0; r < s.size(); r++) {
    cnt[s[r]]++;
    if (cnt[s[r]] <= mp[s[r]]) k++;  // 有用字符
    while (cnt[s[l]] > mp[s[l]]) {
      cnt[s[l]]--;
      l++;
    }
    if (k == t.size()) {
      if (r - l < rr - ll) {
        res = r - l + 1;
        ll = l, rr = r;
      }
    }
  }
  if (res == INT_MAX) {
    return "";
  } else {
    return s.substr(ll, rr - ll + 1);
  }
}
```

在一条环路上有 n 个加油站，其中第 i 个加油站有汽油 `gas[i]` 升。

你有一辆油箱容量无限的的汽车，从第 i 个加油站开往第 i+1 个加油站需要消耗汽油 `cost[i]` 升。你从其中的一个加油站出发，开始时油箱为空。

给定两个整数数组 gas 和 cost ，如果你可以按顺序绕环路行驶一周，则返回出发时加油站的编号，否则返回 -1 。如果存在解，则保证它是唯一的。

```cpp
int canCompleteCircuit(vector<int>& gas, vector<int>& cost) {
  int n = gas.size();
  for (int l = 0, r = 0, sum; l < n; l = r + 1, r = l) {
    sum = 0;
    while (sum + gas[r % n] - cost[r % n] >= 0) {
      if (r - l + 1 == n) {
        return l;
      }
      sum += gas[r % n] - cost[r % n];
      r++;
    }
  }
  return -1;
}
```

有一个只含有 'Q', 'W', 'E', 'R' 四种字符，且长度为 n 的字符串。假如在该字符串中，这四个字符都恰好出现 n/4 次，那么它就是一个「平衡字符串」。给你一个这样的字符串 s，请通过「替换一个子串」的方式，使原字符串 s 变成一个「平衡字符串」。

你可以用和「待替换子串」长度相同的任何其他字符串来完成替换。请返回待替换子串的最小可能长度。如果原字符串自身就是一个平衡字符串，则返回 0。

```cpp
int idx(const char& c) {
  return c - 'A';
}
int balancedString(string s) {
  vector<int> cnt(26);
  for (auto c : s) {
    cnt[idx(c)]++;
  }
  int partial = s.size() / 4;
  int res = s.size();
  auto check = [&]() {
    if (cnt[idx('Q')] > partial || cnt[idx('W')] > partial || cnt[idx('E')] > partial || cnt[idx('R')] > partial) {
      return false;
    }
    return true;
  };
  if (check()) {
    return 0;
  }
  for (int l = 0, r = 0; l < s.size(); l++) {
    while (r < s.size() && !check()) {
      cnt[idx(s[r])]--;
      r++;
    }
    if (!check()) {
      break;
    }
    res = min(res, r - l);
    cnt[idx(s[l])]++;
  }
  return res;
}
```

给定一个正整数数组 nums 和一个整数 k，返回 nums 中 「好子数组」 的数目。
如果 nums 的某个子数组中不同整数的个数恰好为 k，则称 nums 的这个连续、不一定不同的子数组为 「好子数组 」。

```cpp
// 计算不同整数个数小于等于 k 的子数组总数
int atMostK(vector<int>& nums, int k) {
  if (k == 0) return 0;
  int n = nums.size();
  vector<int> freq(n + 1, 0);
  int l = 0, cnt = 0, ans = 0;
  for (int r = 0; r < n; r++) {
    // 进窗口
    if (freq[nums[r]] == 0) {
      cnt++;
    }
    freq[nums[r]]++;
    // 如果不同数字超过 k，收缩左边界
    while (cnt > k) {
      freq[nums[l]]--;
      if (freq[nums[l]] == 0) {
        cnt--;
      }
      l++;
    }
    // 以 r 为右端点，满足条件的子数组个数恰好是当前窗口的长度 (r - l + 1)
    // 比如窗口是 [l, r]，那么 [l, r], [l+1, r], ..., [r, r] 都满足条件
    ans += (r - l + 1);
  }
  return ans;
}
int subarraysWithKDistinct(vector<int>& nums, int k) {
  // 恰好 K 个 = 最多 K 个 - 最多 K-1 个
  return atMostK(nums, k) - atMostK(nums, k - 1);
}
```

给你一个字符串 s 和一个整数 k ，请你找出 s 中的最长子串，要求该子串中的每一字符出现次数都不少于 k 。返回这一子串的长度。如果不存在这样的子字符串，则返回 0。

```cpp
// 在字符串 s 的 [l, r] 区间内寻找符合要求的最长子串长度
int dfs(const string& s, int l, int r, int k) {
  // [控制边界] 如果当前区间长度小于 k，不可能存在满足条件的子串
  if (r - l + 1 < k)
    return 0;
  // [步骤 1] 统计当前区间内每个字符出现的总次数
  // 虽然字符集是 26，但我们只关心 [l, r] 范围内的字符
  int cnt[26] = {0};
  for (int i = l; i <= r; ++i) {
    cnt[s[i] - 'a']++;
  }
  // [步骤 2] 寻找分割点
  // 分割点：该字符在当前区间内出现次数大于 0 且小于 k
  int split = -1;
  for (int i = l; i <= r; ++i) {
    if (cnt[s[i] - 'a'] > 0 && cnt[s[i] - 'a'] < k) {
      split = i;
      break;
    }
  }
  // [步骤 3] 判断是否需要分割
  // 如果找不到任何出现次数 < k 的字符，说明当前整个区间 [l, r] 都符合条件
  if (split == -1) {
    return r - l + 1;
  }
  // [步骤 4] 递归处理
  // 既然 s[split] 这个字符不合格，那么包含它的子串统统不合格
  // 我们以 split 为界，分别检查它的左边和右边
  // 递归左半部分
  int leftMax = dfs(s, l, split - 1, k);
  // [优化] 跳过连续不合格的字符，减少递归深度
  // 比如 "ababa...c...c...ababa"，如果中间连续几个字符都不满 k 次，直接跳过它们
  while (split <= r && cnt[s[split] - 'a'] < k) {
    split++;
  }
  // 递归右半部分
  int rightMax = dfs(s, split, r, k);
  // 返回左右两边找到的最长子串的较大值
  return max(leftMax, rightMax);
}
int longestSubstring(string s, int k) {
  int n = s.length();
  // 如果字符串长度小于 k，直接返回 0，因为任何字符都不可能重复 k 次
  if (n < k)
    return 0;
  return dfs(s, 0, n - 1, k);
}
```

给定一个整数数组 A，坡是元组 `(i, j)`，其中  ` i < j ` 且 ` A[i] <= A[j]`。这样的坡的宽度为 `j - i`。找出 A 中的坡的最大宽度，如果不存在，返回 0 。

```cpp
int maxWidthRamp(vector<int>& nums) {
  int n = nums.size();
  stack<int> st;
  // 阶段一：从左往右，挑选最有潜力的左端点候选人
  // 维护一个值严格单调递减的栈
  for (int i = 0; i < n; i++) {
    // 如果栈为空，或者当前元素 严格小于 栈顶元素，则入栈
    if (st.empty() || nums[i] < nums[st.top()]) {
      st.push(i);
    }
  }
  int max_width = 0;
  // 阶段二：从右往左，贪心地寻找最远的右端点 j
  for (int j = n - 1; j >= 0; j--) {
    // 只要当前的 nums[j] 能够和栈顶元素组成坡 (nums[j] >= nums[st.top()])
    // 我们就计算宽度，并无情地把栈顶弹出（因为它已经找到了属于它的最远 j，潜力已榨干）
    while (!st.empty() && nums[j] >= nums[st.top()]) {
      max_width = max(max_width, j - st.top());
      st.pop();
    }
  }
  return max_width;
}
```
